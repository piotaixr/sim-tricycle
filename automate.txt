/*fichier automate*/
robot{

	etat=normal{
            transition{
		si(piece_trouvee){etat=deplacement_norm;}
            }
            transition{
		sinon{attente(tempos);
		       etat=normal;
		}
            }
	}
	etat=deplacement_norm{
            transition{
		si(case==piece){ ramasser(tempos);
				     etat=porteur;
                }
            }
            transition{
		sinon{aller(piece,tempos);
		      etat=normal;
		}
            }
	}
	etat=porteur{
            transition{
		etat=deplacement_port;
            }
	}
	etat=deplacement_port{
            transition{
		si(case==base){deposer(tempos);
				   etat=normal;
			     }
            }
            transition{
		sinon{aller(base,tempos);
		      etat=deplacement_port;
		}
            }
	}
}
fin
